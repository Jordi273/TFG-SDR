int main(int argc, char **argv){
SdrConfig SdrConf;
VisrtualSdr Sdr;
VirtualSdrError ErrorBuffer;
long sample_rate = 32700000;
int BW = 4000000;
long LOFrec = 868000000;
float att = -50;
PrintSdrErrorCode (LoadBSP(&SdrConf, "ICE.csv"));
PrintSdrErrorCode (SetConnection(&SdrConf, IP, "10.0.0.5"));
PrintSdrErrorCode (ChangeTxChannel(&SdrConf, A));
PrintSdrErrorCode (SetTxFrec(&SdrConf, first, LOFrec));
PrintSdrErrorCode (SetTxBw(&SdrConf, BW));
PrintSdrErrorCode (SetAttenuation(&SdrConf, first, att));
PrintSdrErrorCode (SetFS(&SdrConf, sample_rate));
PrintSdrConfig(&SdrConf);
PrintSdrErrorCode (ChargeConfig(&Sdr, &SdrConf));
float I[2048];
float Q[2048];
for(int t_iteri = 0; i < 2048; i++){
I[t_iter] = cos(2*M_PI*(1000000)*((float)t_iter/sample_rate)));
Q[t_iter] = sin(2*M_PI*(1000000)*((float)t_iter/sample_rate)));
}
PrintSdrErrorCode (TransmitAlways(&Sdr, A, 2048, I, Q));
ErrorBuffer = StartSdr(&Sdr);
if(ErrorBuffer != 0){
PrintSdrErrorCode(VirtualSdrError);
return -1;
}
sleep(10);
PrintSdrErrorCode(StopSdr(&Sdr));
FreeVirtualSdr (&Sdr);
FreeSdrConfig (&SdrConf);
return 0;
}
